217 template <typename T>
218 T & Array<T>::operator [](size_t i) {
219 // This is to provide an opportunity to disable JS-like Array
220 // behavior:
221 //
222 // var a = new Array();
223 // a.length; // -> 0
224 // a[99] = 1;
225 // a.length; // -> 100
226 #ifndef INTOLERANT_TO_OUT_OF_RANGE_INDEXES
227  if (i >= data_.size()) {
228  data_.resize(i + 1);
229  }
230 #endif
231  return data_[i];
232 }

271 Array<T> Array<T>::slice(ssize_t begin, ssize_t end) const {
272  ssize_t length = data_.size();
273 
274  if (
275  begin >= length ||
276  begin < -length ||
277  end >= length ||
278  end < -length
279  ) {
280  return Array<T>();
281  }
282 
283  begin = (length + begin) % length;
284  end = (length + end) % length;
285 
286  if (end < begin) {
287  return Array<T>();
288  }
289 
290  return Array<T>(data_.begin() + begin, data_.begin() + end);
291 }
